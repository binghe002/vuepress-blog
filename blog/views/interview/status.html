# 使(a==1&&a==2&&a==3)条件成立

var a ={
    i:1,
    toString(){
        return a.i++;
        //因为==会进行隐式转换，所以我们重写toString方法就好了
    }
}
if(a==1&&a==2&a==3){
    console.log(1);
}

# 闭包中变量的判断

var b=10;
(function b(b){
    window.b=20;
    console.log(b)
    //作为参数传递进来，所以改变window的变量也无用
    //输出10
})(b)

var b=10;
(function b(b)){
    b=20;
    console.log(b)
    //输出20
}()
